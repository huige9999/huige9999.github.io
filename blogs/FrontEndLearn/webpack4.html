    新建一个html文件，引入上面的js代码，然后在浏览器控制台查看结果，会发现控制台输出了预期的结果

    • 接下来我们再给require函数加上缓存功能，这样可以避免重复运行模块
    (function(modules){
  var moduleExports = {}; // 用于缓存模块导出的结果

  // require函数相当于运行一个模块,得到模块的导出结果
  function require(moduleId) { // moduleId就是模块路径
    if(moduleExports[moduleId]) {
      // 检查是否有缓存
      return moduleExports[moduleId];
    }

    var func = modules[moduleId]; // 得到该模块对象的函数
    var module = {
      exports: {}
    }
    func(module, module.exports, require); // 运行模块
    var result = module.exports; // 得到模块的导出结果
    moduleExports[moduleId] = result; // 缓存模块导出的结果
    return result;
  }

    // 执行入口模块
    require('./src/index.js'); 
})({
    "./src/a.js": function (module, exports) {
      console.log("module a");
      module.exports = "a";
    },
    "./src/index.js": function (module, exports, require) {
      console.log("index module");
      var a = require("./src/a.js");
      console.log(a);
    },
});

    模块代码放在一个函数环境中运行，更具体点是：模块代码变成字符串放到eval函数中执行。

    为什么要放到eval函数中执行？

    • 假设不放到eval函数里面，我们调试代码会定位到webpack合并后的代码中，这样很不利于调试，因为干扰性的代码太多了！

    • eval中的代码是放到另一个环境执行的，这样就能帮我们排除干扰

    • 此外eval中的代码尾端还可以加//# sourceURL=webpack://learn01/./src/index.js?这样的注释，这样我们调试定位的时候可以准确的定位到模块代码的位置。
