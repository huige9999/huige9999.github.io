    # 相关面试题

    # babel是什么,它在JavaScript开发中的作用是什么?

    babel是一个JavaScript编译器,主要用于将现代JavaScript代码(包括ES6及更高版本的语法),转换为向后兼容的JavaScript代码, 以确保在旧版本的浏览器或环境中也能运行,它的作用包括：

    1. 语法转换：将现代JavaScript语法(如箭头函数、类、模板字符串等)转换为ES5兼容的语法。
    2. Polyfills：通过使用工具如@babel/polyfill，可以为缺失的JavaScript功能(如:Promise、Array.includes等)添加兼容性支持。
    3. 插件化架构: babel使用插件来扩展其功能,可以根据项目需求加载不同插件进行定制。
    4. 代码优化和压缩: 虽然主要用于语法转换,babel也可以与其他工具(如Terser)集成,以优化和压缩代码。

    # 解释babel插件和预设的区别，并举例说明如何在项目中使用它们

    插件和预设都用于配置babel的行为,但它们有不同的用途和方式。

    • 插件(Plugins): 插件时babel的基本转换单元,负责特定的代码转换任务。例如,将箭头函数转换为普通函数的插件@babel/plugin-transform-arrow-functions。
    • 预设(Presets): 预设时多个插件的集合,用于简化配置。例如,@babel/preset-env预设包含了一系列插件,能够根据目标环境自动选择和加载所需要的插件,以实现ES6+到ES5的转换。

    使用示例 在项目中,通常会创建一个.babelrc文件进行配置：

    {
 'presets': ['@babel/preset-env'],
 'plugins': ['@babel/plugin-transform-runtime']
}

    # 什么是抽象语法树(AST),它在babel中的作用是什么?

    抽象语法树(AST)是源代码的抽象语法结构的树状表示,其中每个节点都代表源代码中的一种结构,在编译器和babel的工作流程中,AST扮演着重要的角色。

    AST在babel中的作用

    1. 代码解析(Parsing): Babel首先解析源代码,生成AST。解析器(如Babylon)将源代码转换为AST,这是源代码转换的第一步。
    2. 代码转换(Transformation): Babel插件对AST进行遍历和修改,根据需要转换特定的语法结构。例如,箭头函数插件会找到AST中的箭头函数节点,并将其转换为普通函数节点。
    3. 代码生成(Code Generation): 在完成所有转换后,Babel将修改后的AST转换回JavaScript代码。这一步通过生成器(如Babel生成器)实现的

const CalcutatorInterceptors = {}

class Caculator {
    plugins = new Map();
    constructor() {
     // 计算逻辑的名称和计算逻辑的函数
     // plugins = new Map();
     // 约定一下插件的协议
    /**
     * {
     *  name: 'add',
     *  handler: function(a,b){return a+b}    
     * }
     */
    }
    // 注册插件的逻辑
    use(plugin) {
        this.plugins.set(plugin.name, plugin.handler);
    }
    // 执行插件的逻辑
    invoke(name,...args) {
        if(!this.plugins.has(name)) {
            throw new Error(`plugin ${name} not found`);
        }
        // 前置处理
        CalcutatorInterceptors.enter?.[name] && CalcutatorInterceptors.enter[name](name,...args)
        // 执行核心逻辑
        const result = this.plugins.get(name)(...args);
        // 后置处理
        CalcutatorInterceptors.exit?.[name] && CalcutatorInterceptors.exit[name](name, result,...args)
        return result;
    }

}

                noNewArrows,
                specCompliant: !noNewArrows
               });
            }
        }
    }
});

    # babel编译过程

    转换型编译器的执行流程

    • tokenizer: 词法分析，将代码分割成一个个token
    • parser: 语法分析，将token转换成AST
    • transformer: 转换器，对AST进行转换，比如将箭头函数转换为普通函数
    • generator: 代码生成器，将AST转换成代码

    # 编译原理

    • 词法分析：将代码转换成单词流，称为"词法单元"(tokens)，每个词法单元包含一个标识符和一个属性值，比如变量名、数字、操作符等等 词法分析生成token的方法有2种：1. 正则表达式；2. 状态机
    [const, a, =, 1]
    1
    • 语法分析： 将词法单元流转换成抽象语法树(AST)，也就是标记所构成的数据结构，表示源代码的结构和规则
    • 语义分析： 在AST上执行类型检查，作用域检查等操作，确保代码的正确性
    • 代码生成： 基于AST生成目标代码,包括优化代码结构、生成代码文本、进行代码压缩等。

    // 1. token定义
// 用于提取公式中的函数名
const FN_NAME_TOKEN = /[a-zA-Z]/
// 用于提取数字
const NUMBER_TOKEN = /\d/
// 用于提取左括号
const LEFT_PAREN_TOKEN = /\(/
// 用于提取右括号
const RIGHT_PAREN_TOKEN = /\)/
// 用于提取“,”
const COMMA_TOKEN = /,/

// 2. tokenizer
function tokenizer(code) {
    const tokens = [];
    let current = 0; // 当前指针
    while (current < code.length) {
        let char = code[current];

        // 先匹配数字
        if (NUMBER_TOKEN.test(char)) {
            // 贪婪匹配数字
            let value = '';
            while (NUMBER_TOKEN.test(char)) {
                value += char; // 当前字符存起来
                current++; // 指针后移
                char = code[current]; // 取下一个字符
            }
            // 将匹配到的数字token加入到tokens中
            tokens.push({ type: 'number', value: parseInt(value) });
            continue;
        }

        // 再匹配函数名
        if(FN_NAME_TOKEN.test(char)) {
            let value = '';
            // 贪婪匹配函数名
            while (FN_NAME_TOKEN.test(char)) {
                value += char; // 当前字符存起来
                current++; // 指针后移
                char = code[current]; // 取下一个字符
            }
            // 将匹配到的函数名token加入到tokens中
            tokens.push({ type: 'function', value });
            continue;
        }

        // 匹配括号和逗号
        if(LEFT_PAREN_TOKEN.test(char) || RIGHT_PAREN_TOKEN.test(char) || COMMA_TOKEN.test(char)) {
            tokens.push({
                type: char,
                value: char
            })
            current++; // 指针后移
            continue;
        }

        // 处理空格
        if(char === " ") {
            current++; // 指针后移
            continue;
        }

        // 其他情况则抛出异常
        throw new TypeError('I dont know what this charactor is:'+ char);



        current++; // 指针后移
    }


    return tokens;
}

const tokens = tokenizer('Add(1,MINUS(2,3))');

    // 3.parser
function parser(tokens) {

    function traverse() {
        let token = tokens[current];

        // 处理数字
        if(token.type === 'number') {
            current++;
            return {
                type: 'NumberLiteral',
                value: token.value
            }
        }

        // 处理函数
        if(token.type === 'function') {
            current++;
            // 处理基础节点
            let node = {
                type: 'CallExpression',
                name: token.value,
                params: []
            }
            // 处理函数参数
            token = tokens[current];
            // 匹配"("
            while(LEFT_PAREN_TOKEN.test(token.value) || COMMA_TOKEN.test(token.value)) {
                node.params.push(traverse());
                token = tokens[current];
                // 如果匹配到的是","则继续循环
                if(COMMA_TOKEN.test(token.value)) {
                    current++; // 跳过逗号
                }
            }
            current++; // 跳过")"
            return node;
        }


    }

    // 基础的AST结构
     let ast = {
        type: 'program',
        body: []
     }

     let current = 0; // 当前指针

     // 一个一个提取处理token
     while(current < tokens.length) {
        const body = traverse();
        body && ast.body.push(body);
     }
     

    return ast;
}


const ast = parser(tokens);

    // 4.interpreter
// 后续遍历ast完成计算
function interpreter(ast) {
   const operators = {
      Add: (a,b) => a + b,
      Subtract: (a,b) => a - b,
      Multiply: (a,b) => a * b,
      Divide: (a,b) => a / b,
   }

    function traverse(node) {
        switch(node.type) {
            case 'NumberLiteral':
                return node.value
            case 'CallExpression':
                const args = node.params.map(traverse);
                const operator = operators[node.name];
                return operator(...args);
        }
    }

    return traverse(ast.body[0]);
}

interpreter(ast);
